91 inline int XorHandler::check(const Arg *a) {
92  // iterate over each XOR list
93  for (int i = 0; static_cast<unsigned int>(i) < _orList.size(); i++) {
94  // if the XOR list contains the arg..
95  ArgVectorIterator ait = std::find(_orList[i].begin(), _orList[i].end(), a);
96  if (ait != _orList[i].end()) {
97  // first check to see if a mutually exclusive switch
98  // has not already been set
99  for (ArgVectorIterator it = _orList[i].begin(); it != _orList[i].end();
100  it++)
101  if (a != (*it) && (*it)->isSet())
102  throw (CmdLineParseException(
103  "Mutually exclusive argument already set!", (*it)->toString()));
104 
105  // go through and set each arg that is not a
106  for (ArgVectorIterator it = _orList[i].begin(); it != _orList[i].end();
107  it++)
108  if (a != (*it))
109  (*it)->xorSet();
110 
111  // return the number of required args that have now been set
112  if ((*ait)->allowMore())
113  return 0;
114  else
115  return static_cast<int>(_orList[i].size());
116  }
117  }
118 
119  if (a->isRequired())
120  return 1;
121  else
122  return 0;
123 }
