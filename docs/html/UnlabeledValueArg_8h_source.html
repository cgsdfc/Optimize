239 template <class T>
240 bool UnlabeledValueArg<T>::processArg(int *i, std::vector<std::string> &args) {
241 
242  if (_alreadySet)
243  return false;
244 
245  if (_hasBlanks(args[*i]))
246  return false;
247 
248  // never ignore an unlabeled arg
249 
250  _extractValue(args[*i]);
251  _alreadySet = true;
252  return true;
253 }

267 template <class T>
268 std::string UnlabeledValueArg<T>::longID(const std::string &val) const {
269  static_cast<void>(val); // Ignore input, don't warn
270 
271  // Ideally we would like to be able to use RTTI to return the name
272  // of the type required for this argument. However, g++ at least,
273  // doesn't appear to return terribly useful "names" of the types.
274  return std::string("<") + _typeDesc + ">";
