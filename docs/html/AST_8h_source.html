73 struct DeleteAST {
78  template <typename Iterator>
79  static void apply(Iterator first, Iterator last) {
80  std::for_each(first, last, DeleteAST());
81  }
83  template <typename AstT> static void apply(const std::vector<AstT *> &List) {
84  return apply(List.begin(), List.end());
85  }
87  static void apply(AST *A) {
88  if (A)
89  A->deleteAST();
90  }
92  void operator()(AST *A) const { return apply(A); }
93 };
